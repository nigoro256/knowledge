== 一時ファイルを作成
[source, sh]
----
readonly tmpfile=$(mktemp)

function rm_tmpfile() {
  [[ -f "$tmpfile" ]] && rm -f "$tmpfile"
}
trap rm_tmpfile EXIT
trap 'trap - EXIT; rm_tmpfile; exit -1' INT PIPE TERM
----
